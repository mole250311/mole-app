CREATE TABLE user_profiles (
    user_id           INT PRIMARY KEY,                   -- users 테이블과 1:1 매핑
    username          VARCHAR(30) NOT NULL,              -- 사용자 ID (조회 편의용, 중복 가능)
    student_id        VARCHAR(20) NOT NULL UNIQUE,       -- 학번
    school_name       VARCHAR(100),                      -- 학교명
    department_name   VARCHAR(100),                      -- 학과명
    last_chapter_id   INT,                               -- 마지막 종료 챕터 ID
    last_page_number  INT,                               -- 마지막 종료 페이지 번호
    updated_at        DATETIME DEFAULT CURRENT_TIMESTAMP 
                          ON UPDATE CURRENT_TIMESTAMP,   -- 갱신 시각

    FOREIGN KEY (user_id) REFERENCES users(user_id)
        ON DELETE CASCADE
);
